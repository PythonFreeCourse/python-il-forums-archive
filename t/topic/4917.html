GOOD = 2
CAKE = 3
ERROR = 4


def normalize_ripeness_level(banana_ripeness):
    if not banana_ripeness.isdecimal():
        return ERROR
    
    if 1 <= banana_ripeness <= 4:
        return RIPE
    elif 5 <= banana_ripeness <= 7:
        return GOOD
    elif 8 <= banana_ripeness <= 10:
        return CAKE
    return ERROR
