def read_file(path):
    try:
        princess = open(path, 'r')
        text = princess.read()
    except (FileNotFoundError, PermissionError) as err:
        print(f"Can't find file '{path}'.\n{err}.")
        text = None
    else:
        princess.close()
    finally:
        return text


print(read_file('resources/castle.txt3'))
