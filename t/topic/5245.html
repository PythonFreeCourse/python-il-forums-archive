def maximum(list_of_numbers):
    if len(list_of_numbers) == 1:
        return list_of_numbers[0]
    
    if list_of_numbers[0] > list_of_numbers[1]:
        list_of_numbers.pop(1)
        return maximum(list_of_numbers)
    list_of_numbers.pop(0)
    return maximum(list_of_numbers)
