def maximum(lst):
i = len(lst) -1
if i == 0:
return lst[i]
res = lst.copy()
res.pop()
res = maximum(res)
#another option is - return max(res,lst[i])
if res > lst[i]:
return res
else:
return lst[i]

def maximum(lis):
    if len(lis) == 1:
        return lis[0]
    if lis[0] > lis[1]:
        return maximum(lis[0:1] + lis[2:])
    return maximum(lis[1:])
