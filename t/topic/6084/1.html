def find_pythagorean_triples(upper_bound=10_000):
    pythagorean_triples = []
    for c in range(3, upper_bound):
        for b in range(2, c):
            for a in range(1, b):
                if a ** 2 + b **2 == c ** 2:
                    pythagorean_triples.append((a, b, c))
    return pythagorean_triples
