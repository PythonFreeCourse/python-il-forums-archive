def random_date(d1, d2):
d1 = d1.split(’-’)
d2 = d2.split(’-’)
d1 = datetime.date(int(d1[0]), int(d1[1]), int(d1[2]))
d2 = datetime.date(int(d2[0]), int(d2[1]), int(d2[2]))
days_between = (d2 - d1).days
date = d1 + datetime.timedelta(days=random.randint(0, days_between))
return date
