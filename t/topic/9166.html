def level_ten(exp):
lvl = exp // 50
return lvl

def level_ten_up(exp):
lvl = ((exp - 500) // 100) + 10
return lvl

def what_level(exp):
if exp <= 500:
lvl = level_ten(exp)
elif exp > 500:
lvl = level_ten_up(exp)
return lvl

def carry(level):
if level % 2 == 1:
items = level // 2 + 1
else:
items = level // 2
return items

print(carry(what_level(1337)))
