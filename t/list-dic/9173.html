מתודה א: – מיון בעזרת רשימה - צורה לא יעילה

my_list = ['a', 'a', 'a', 'b', 'b', 'c']
new_dic = {}
while my_list:
    counting = 0
    current = my_list[0]
    while counting < len(my_list) and my_list[counting] == current:
        counting += 1
    new_dic[current] = counting
    my_list = my_list[ counting :]
print(new_dic)
# {'a': 3, 'b': 2, 'c': 1}

מתודה ב: – מיון בעזרת מילון

my_list = ['a', 'a', 'a', 'b', 'b', 'c']
new_dic = {}
for word in my_list:
    if word in new_dic:
        new_dic[word] += 1
    else:
        new_dic[word] = 1
print(new_dic)
# {'a': 3, 'b': 2, 'c': 1}
